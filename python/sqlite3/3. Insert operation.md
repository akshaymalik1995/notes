```python
import sqlite3

db = sqlite3.connect('students.db')

class Student:
    def __init__(self, id, name, age, dob, lang):
        self.id = id 
        self.name = name 
        self.age = age 
        self.dob = dob 
        self.lang = lang

    def insert(self):
        db.execute(f'''
        INSERT INTO STUDENT (ID, NAME, AGE, DOB, LANGUAGE)
        VALUES({self.id}, '{self.name}', {self.age}, {self.dob}, '{self.lang}')

        ''')

        db.commit()


f = open("students.txt", "r")
lines = f.readlines()
for line in lines:
    line = line.strip().split(",")
    id = int(line[0])
    name = line[1]
    age = int(line[2])
    dob = int(line[3])
    lang = line[4]
    student = Student(id, name, age, dob, lang)
    student.insert()

db.close()
```
