## Setting up Context API

1. In the `src` folder, create a file `Context.js`

```jsx
import React from "react"
const context = React.createContext();
export default context;

```

2. Create a file `Provider.js`

```jsx
import context from "./Context";
import React from "react";
class Provider extends React.Component {
    state = {
        todos : [
            {
                id : 1,
                task : "Complete your homework",
                finished : false,
                created : "09:12:34"
            }, 
            {
                id : 2,
                task : "Play Tennis",
                finished : false,
                created : "09:13:34"
            }
        ]
    }

    render () {
        return (
            <context.Provider value={this.state}>
                {this.props.children}
            </context.Provider>

        )
    }
}

export default Provider

```


3. In `App.js`, Import `Provider` and wrap the App Component with it.

```jsx
import { Input } from "./components/Input";
import { Todos } from "./components/Todos";
import Provider from "./Provider";

function App() {

  return (
    <Provider>
      <div className="App container">
      <h1 className='text-center m-5'>Todo</h1>
      <Input />
      <Todos />
    </div>

    </Provider>
    
  );
}

export default App;

```

4. Accessing the state in a component

**Todos.js**

```jsx
import React from 'react'
import { Todo } from './Todo'
import context from '../Context'
export const Todos = () => {
    return (
    <context.Consumer>
        {
            (state) => {
                return (
                    <div className="my-5">
                        <ul class="list-group">
                            {state.todos.map(todo => (
                                    <Todo id={todo.id} todo={todo} />))}
                        </ul>
                    </div>
                )
            }
        }
    </context.Consumer>
    )
    
}

```
